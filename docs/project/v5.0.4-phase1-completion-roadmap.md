```python
def _print_metadata_section(self, result: Dict[str, Any]) -> None:
    """Print enhanced metadata section with popularity and links."""
    metadata = result.get("metadata", {})
    statistics = metadata.get("statistics", {})

    # Only show if we have interesting metadata
    if not (statistics or metadata.get("repository_url")):
        return

    console.print("\n[bold]ğŸ“Š Extension Metadata[/bold]")
    console.print("â”€" * 50)

    # Install count
    installs = statistics.get("installs")
    if installs:
        formatted_installs = self._format_install_count(installs)
        console.print(f"Popularity: {formatted_installs} installs")

    # Rating
    rating = statistics.get("rating")
    rating_count = statistics.get("rating_count")
    if rating:
        stars = "â˜…" * int(rating) + "â˜†" * (5 - int(rating))
        rating_text = f"{rating:.1f} {stars}"
        if rating_count:
            rating_text += f" ({rating_count:,} ratings)"
        console.print(f"Rating: {rating_text}")

    # Repository
    repo_url = metadata.get("repository_url")
    if repo_url:
        console.print(f"Repository: [link]{repo_url}[/link]")

    # License
    license_name = metadata.get("license")
    if license_name:
        console.print(f"License: {license_name}")

    # Keywords/Categories
    keywords = metadata.get("keywords", [])
    if keywords and isinstance(keywords, list):
        console.print(f"Keywords: {', '.join(keywords[:5])}")  # First 5

    categories = metadata.get("categories", [])
    if categories and isinstance(categories, list):
        console.print(f"Categories: {', '.join(categories)}")

def _format_install_count(self, count: int) -> str:
    """Format install count with appropriate units."""
    if count >= 1_000_000:
        return f"{count / 1_000_000:.1f}M"
    elif count >= 1_000:
        return f"{count / 1_000:.1f}K"
    else:
        return str(count)
```

## Risks & Mitigations

### Risk 1: Chart.js File Size
**Severity**: LOW
**Probability**: HIGH (guaranteed)
**Impact**: HTML reports increase from ~50KB to ~250KB

**Mitigation**:
- Chart.js minified is ~200KB (acceptable for self-contained reports)
- Consider lazy loading (load only if score contributions present)
- Alternative: Defer chart to v5.0.5, ship other enhancements in v5.0.4

### Risk 2: Browser Compatibility
**Severity**: LOW
**Probability**: LOW
**Impact**: Chart may not render in very old browsers

**Mitigation**:
- Chart.js supports IE11+ (broad compatibility)
- Table display (v5.0.3) remains as fallback
- Chart is enhancement, not requirement

### Risk 3: Security Notes Verbosity
**Severity**: LOW
**Probability**: MEDIUM
**Impact**: Some extensions may have 10+ notes (cluttered reports)

**Mitigation**:
- Limit display to first 10 notes with "show more" option
- Or: Collapsible section (JavaScript toggle)
- Or: Summary count ("5 security concerns identified")

### Risk 4: Missing Metadata
**Severity**: LOW
**Probability**: MEDIUM
**Impact**: Some extensions lack install counts or ratings

**Mitigation**:
- Graceful degradation (hide missing fields)
- Already handled in v4.0.0 parser (NULL-safe)
- Test coverage includes missing data scenarios
